Spring Module v2.1

The industry standard for smooth camera sway, viewmodel movement, and UI animations.

local Spring = {}
Spring.__index = Spring
function Spring.new(mass, force, damping, speed)
    return setmetatable({
        Target = Vector3.new(), Position = Vector3.new(), Velocity = Vector3.new(),
        Mass = mass or 1, Force = force or 50, Damping = damping or 4, Speed = speed or 4
    }, Spring)
end
function Spring:Update(dt)
    local d = self.Target - self.Position
    local f = d * self.Force
    local df = self.Velocity * self.Damping
    local a = (f - df) / self.Mass
    self.Velocity = self.Velocity + a * dt * self.Speed
    self.Position = self.Position + self.Velocity * dt * self.Speed
    return self.Position
end
return Spring
Navigation & AI

Targeting Research Module

Advanced logic for team-aware target selection and smooth camera interpolation (Spring-based). No 'tweaking'.

-- Targeting Module (Educational)
local Targeting = {}
local Spring = require(SpringModule)
local Smooth = Spring.new(1, 12, 4, 1)

function Targeting.GetClosest(maxDist)
    local target, dist = nil, maxDist or math.huge
    for _, p in ipairs(game.Players:GetPlayers()) do
        if p ~= game.Players.LocalPlayer and p.Team ~= game.Players.LocalPlayer.Team then
            local char = p.Character
            if char and char:FindFirstChild("Head") then
                local d = (char.Head.Position - game.Players.LocalPlayer.Character.Head.Position).Magnitude
                if d < dist then target = char.Head; dist = d end
            end
        end
    end
    return target
end

function Targeting.SmoothRotate(targetCFrame, dt)
    Smooth.Target = targetCFrame.Position
    local p = Smooth:Update(dt)
    return CFrame.new(workspace.CurrentCamera.CFrame.Position, p)
end

return Targeting

    task.wait(dt) -- Wait for precise timing

Advanced NPC Path AI

Functional pathfinding with dynamic target tracking and avoidance.

local PFS = game:GetService("PathfindingService")
local function WalkTo(npc, pos)
    local path = PFS:CreatePath({AgentCanJump = true})
    path:ComputeAsync(npc.HumanoidRootPart.Position, pos)
    for _, w in ipairs(path:GetWaypoints()) do
        npc.Humanoid:MoveTo(w.Position)
        if w.Action == Enum.PathfindingWaypointAction.Jump then
            npc.Humanoid.Jump = true
        end
        npc.Humanoid.MoveToFinished:Wait()
    end
end
